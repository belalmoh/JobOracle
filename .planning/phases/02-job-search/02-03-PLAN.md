        location?: string;
        date_from?: string;
        sort?: string;

      }) => {
        const query = new URLSearchParams(params as Record<string, string>).toString();
        return fetchAPI(`/jobs${query ? `?${query}` : ""}`);
      },
      get: async (id: number) => fetchAPI(`/jobs/${id}`),
      score: async (jobId: number) =>

          method: "POST",

      scoreAll: async (jobIds?: number[]) =>
